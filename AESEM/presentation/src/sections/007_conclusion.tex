\section{Conclusion}

\begin{frame}
    \frametitle{Conclusion}

    \note<1->[item]{
        Much of the literature \begin{itemize}
            \item Concerns itself with euclidean graphs
        \end{itemize}
    }
    \note<2->[item]{
        Two algorithms \begin{itemize}
            \item Tsiligiridis S-Algorithm
            \item Szwarc and Boryczka's Algorithm
        \end{itemize}
    }
    \note<3->[item]{
        Possibility of generalization varies \begin{itemize}
            \item S-Algorithm: Can likely be reduced to triangle inequality
            \item Szwarc-Boryczka: Need to remove heuristics to drop euclidean requirement
        \end{itemize}
    }
    \note<4->[item]{
        Impact on performance and solution quality \begin{itemize}
            \item Can hardly be evaluated a-priori
            \item Requires experimental evaluation
        \end{itemize}
    }

    \begin{itemize}
        \item<1-> Much of the literature: Euclidean Graphs or other constraints
        \item<2-> Seen two algorithms
        \item<3-> Possibility for generalization varies
        \item<4-> Impact on performance to be evaluated experimentally
    \end{itemize}    

\end{frame}
